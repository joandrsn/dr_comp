OBJECT Report 1140 Update Cost Acctg. Dimensions
{
  OBJECT-PROPERTIES
  {
    Date=15-09-15;
    Time=12:00:00;
    Version List=NAVW19.00;
  }
  PROPERTIES
  {
    CaptionML=[DAN=Opdater dimensioner for omkostningsregnskab;
               ENU=Update Cost Acctg. Dimensions];
    ProcessingOnly=Yes;
    OnPostReport=BEGIN
                   IF UpdateDimension THEN BEGIN
                     IF NOT CONFIRM(Text001,TRUE) THEN
                       ERROR('');

                     CostAccSetup.VALIDATE("Cost Center Dimension",NewCCDimension);
                     CostAccSetup.VALIDATE("Cost Object Dimension",NewCODimension);
                     CostAccSetup.MODIFY(TRUE);
                   END;
                 END;

  }
  DATASET
  {
  }
  REQUESTPAGE
  {
    PROPERTIES
    {
      OnOpenPage=BEGIN
                   CostAccSetup.GET;
                   GetInitialDimensions;
                 END;

    }
    CONTROLS
    {
      { 1   ;    ;Container ;
                  ContainerType=ContentArea }

      { 2   ;1   ;Group     ;
                  CaptionML=[DAN=Indstillinger;
                             ENU=Options];
                  GroupType=Group }

      { 3   ;2   ;Field     ;
                  Name=CostCenterDimension;
                  CaptionML=[DAN=Omkostningsstedsdimension;
                             ENU=Cost Center Dimension];
                  SourceExpr=NewCCDimension;
                  TableRelation=Dimension;
                  OnValidate=BEGIN
                               IF NewCCDimension = '' THEN BEGIN
                                 GetInitialDimensions;
                                 UpdateDimension := FALSE;
                                 ERROR(Text003,CostAccSetup.FIELDCAPTION("Cost Center Dimension"));
                               END;
                               IF NewCCDimension = NewCODimension THEN BEGIN
                                 GetInitialDimensions;
                                 UpdateDimension := FALSE;
                                 ERROR(Text002);
                               END;

                               IF CostAccSetup."Cost Center Dimension" <> NewCCDimension THEN
                                 UpdateDimension := TRUE;
                             END;
                              }

      { 4   ;2   ;Field     ;
                  Name=CostObjectDimension;
                  CaptionML=[DAN=Omkostningsemnedimension;
                             ENU=Cost Object Dimension];
                  SourceExpr=NewCODimension;
                  TableRelation=Dimension;
                  OnValidate=BEGIN
                               IF NewCODimension = '' THEN BEGIN
                                 GetInitialDimensions;
                                 UpdateDimension := FALSE;
                                 ERROR(Text003,CostAccSetup.FIELDCAPTION("Cost Object Dimension"));
                               END;
                               IF NewCCDimension = NewCODimension THEN BEGIN
                                 GetInitialDimensions;
                                 UpdateDimension := FALSE;
                                 ERROR(Text002);
                               END;

                               IF CostAccSetup."Cost Object Dimension" <> NewCODimension THEN
                                 UpdateDimension := TRUE;
                             END;
                              }

    }
  }
  LABELS
  {
  }
  CODE
  {
    VAR
      CostAccSetup@1000 : Record 1108;
      NewCCDimension@1001 : Code[20];
      NewCODimension@1002 : Code[20];
      UpdateDimension@1003 : Boolean;
      Text001@1004 : TextConst 'DAN=Før du ændrer den tilknyttede dimension for finansposter, skal du sikre dig, at alle finansposter, der bruger den tidligere definerede dimension, er overført til omkostningsregnskabet. \\Vil du fortsætte?;ENU=Before you change the corresponding dimension on G/L entries, make sure all G/L entries using the previously defined dimension have been transferred to Cost Accounting. \\Do you want to proceed?';
      Text002@1005 : TextConst 'DAN=Dimensionsværdierne for omkostningssted og omkostningsemne må ikke være de samme.;ENU=The dimension values for cost center and cost object cannot be same.';
      Text003@1006 : TextConst 'DAN=%1 skal udfyldes. Angiv en værdi.;ENU=%1 must be filled in. Enter a value.';

    LOCAL PROCEDURE GetInitialDimensions@1();
    BEGIN
      NewCCDimension := CostAccSetup."Cost Center Dimension";
      NewCODimension := CostAccSetup."Cost Object Dimension";
    END;

    BEGIN
    END.
  }
  RDLDATA
  {
  }
}
